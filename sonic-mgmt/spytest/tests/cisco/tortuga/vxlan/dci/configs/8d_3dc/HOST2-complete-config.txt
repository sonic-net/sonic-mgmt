echo "PS1=\"HOST2$ \"" >> ~/.bashrc
sudo tee /etc/netplan/01-netcfg.yaml > /dev/null << 'EOF'
network:
  version: 2
  renderer: networkd
  ethernets:
    eth1:
      dhcp4: no
    eth2:
      dhcp4: no
  bonds:
    bond0:
      dhcp4: no
      interfaces:
        - eth1
        - eth2
      parameters:
        mode: 802.3ad
        transmit-hash-policy: layer3+4
        mii-monitor-interval: 100
        lacp-rate: fast
      addresses:
        - 10.212.10.2/24
        - 10:212:10::2/64 
      gateway4: 10.212.10.10
EOF

sudo netplan apply
sudo ifconfig eth1 hw ether 00:00:00:00:00:02
sudo ifconfig eth2 hw ether 00:00:00:00:00:02
sudo ifconfig bond0 hw ether 00:00:00:00:00:02
ping -c 1 10.212.10.10
