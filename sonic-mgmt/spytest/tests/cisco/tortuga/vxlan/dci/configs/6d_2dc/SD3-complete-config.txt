# SD3 Complete Configuration (DC2 Spine + DCI Gateway)
# Combines original config with DCI functionality

# Set hostname first
sudo config hostname SD3-DC2-GW1

# ================================
# SONiC Configuration
# ================================

# Intra-DC interfaces (IPv6 link-local only)
sudo config interface ipv6 enable use-link-local-only Ethernet1_1
sudo config interface ipv6 enable use-link-local-only Ethernet1_2

# Inter-DC interfaces (IPv4 + IPv6 link-local)
sudo config interface startup Ethernet1_8
sudo config interface startup Ethernet1_9
sudo config interface startup Ethernet1_10
sudo config interface ip add Ethernet1_8 172.16.1.0/31
sudo config interface ip add Ethernet1_9 172.16.1.2/31
sudo config interface ip add Ethernet1_10 172.16.3.0/31

# Loopback Interfaces
sudo config interface ip add Loopback0 fd28::233:d0c6:fed3/128
sudo config interface ip add Loopback0 200.200.200.3/32

# DCI Loopback (IPv4 for inter-DC)
sudo config interface ip add Loopback100 30.30.30.30/32
sudo config interface ip add Loopback11 102.102.102.102/32

# VXLAN Configuration
sudo config vlan add 10
sudo config vxlan add vxlan-local fd28::233:d0c6:fed3
sudo config vxlan evpn_nvo add NVO vxlan-local
sudo config vxlan add vxlan-remote 102.102.102.102
sudo config vxlan evpn_nvo add NVO vxlan-remote
sudo config vxlan map add vxlan-local 10 7010
sudo config vxlan map add vxlan-remote 10 5011
sudo counterpoll tunnel enable
sudo counterpoll tunnel interval 2000

# ================================
# FRR BGP Configuration
# ================================

vtysh
debug zebra kernel
debug zebra kernel msgdump
debug zebra rib
debug zebra rib detailed
debug zebra vxlan
debug zebra evpn mh mac
debug bgp evpn mh route
debug bgp evpn mh es
debug bgp zebra
debug zebra evpn mh neigh
debug zebra dplane detailed
configure terminal
log stdout
log syslog
hostname SD3-DC2-GW1

router bgp 65003
bgp router-id 200.200.200.3
no bgp ebgp-requires-policy
no bgp default ipv4-unicast
bgp disable-ebgp-connected-route-check
bgp bestpath as-path multipath-relax

# Peer Groups for Intra-DC (IPv6)
neighbor TRANSIT_DC peer-group
neighbor TRANSIT_DC remote-as external
neighbor TRANSIT_DC ebgp-multihop 1

# Peer Groups for Inter-DC (IPv4)
neighbor TRANSIT_WAN peer-group
neighbor TRANSIT_WAN remote-as external
neighbor TRANSIT_WAN ebgp-multihop 1

# Peer Groups for Intra-DC EVPN
neighbor OVERLAY_DC peer-group
neighbor OVERLAY_DC remote-as external
neighbor OVERLAY_DC disable-connected-check
neighbor OVERLAY_DC ebgp-multihop 255
neighbor OVERLAY_DC update-source Loopback0
neighbor OVERLAY_DC domain vxlan-local
neighbor OVERLAY_DC reoriginate vxlan-remote

# Peer Groups for Inter-DC EVPN
neighbor OVERLAY_WAN peer-group
neighbor OVERLAY_WAN remote-as external
neighbor OVERLAY_WAN disable-connected-check
neighbor OVERLAY_WAN ebgp-multihop 255
neighbor OVERLAY_WAN update-source Loopback100
neighbor OVERLAY_WAN domain vxlan-remote
neighbor OVERLAY_WAN reoriginate vxlan-local

# Intra-DC Neighbors (IPv6)
neighbor fd28::233:d0c6:fed7 peer-group OVERLAY_DC
neighbor fd28::233:d0c6:fed7 description "dc2-leaf1-sd7-evpn"

# Intra-DC Transit Neighbor (single interface to SD7 with interface name in description)
neighbor Ethernet1_2 interface peer-group TRANSIT_DC
neighbor Ethernet1_2 description "dc2-leaf1-sd7-1_2"

# Inter-DC Neighbor (interface-based connection with interface name in description)
neighbor Ethernet1_8 interface peer-group TRANSIT_WAN
neighbor Ethernet1_8 description "dc1-gw1-sd1-1_8"
neighbor 10.10.10.10 peer-group OVERLAY_WAN
neighbor 10.10.10.10 description "dc1-gw1-evpn-sd1"
neighbor 20.20.20.20 peer-group OVERLAY_WAN
neighbor 20.20.20.20 description "dc1-gw2-evpn-sd2"

# Address Families
address-family ipv4 unicast
redistribute connected
neighbor TRANSIT_WAN activate
neighbor OVERLAY_WAN activate
network 200.200.200.3/32
network 10.100.2.3/32
exit-address-family

address-family ipv6 unicast
redistribute connected
neighbor TRANSIT_DC activate
exit-address-family

address-family l2vpn evpn
no use-es-l3nhg
neighbor OVERLAY_DC activate
neighbor OVERLAY_WAN activate
advertise-all-vni
advertise ipv4 unicast
advertise ipv6 unicast
exit-address-family
exit
exit
